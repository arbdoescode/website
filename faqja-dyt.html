#include <stdlib.h>

int main()
{

    }
   return (EXIT_SUCCESS);
}
int show_status (double percent)
{
  int x;
    for(x = 0; x <percent/3; x++)
    {
      printf("|");
    }
      printf(" %.2f%%\r", percent);
      fflush(stdout);
      system("sleep 0.25");
    return(EXIT_SUCCESS);
}

Note:At the double value "numtasks" you display the number of ticks your progress bar will have, at the value which divides the percent you determine the bars per persentage you want in your progress bar, and in the end at system the value you put in sleep will determine the speed of progress bar.
